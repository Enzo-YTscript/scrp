-- Fungsi untuk mendapatkan daftar nama NPC dari setiap zona
local function getNPCList()
    local npcs = {}
    local armWrestling = workspace:WaitForChild("GameObjects"):WaitForChild("ArmWrestling")

    for _, zone in pairs(armWrestling:GetChildren()) do
        local npcParent = zone:FindFirstChild("NPC")
        if npcParent then
            for _, npc in pairs(npcParent:GetChildren()) do
                table.insert(npcs, "Zone: " .. zone.Name .. " || NPC Name: " .. npc.Name)
            end
        end
    end
    
    -- Mengurutkan daftar NPC berdasarkan zona dan nama NPC
    table.sort(npcs, function(a, b)
        local zoneA, nameA = a:match("Zone: (.+) || NPC Name: (.+)")
        local zoneB, nameB = b:match("Zone: (.+) || NPC Name: (.+)")
        
        if tonumber(zoneA) and tonumber(zoneB) then
            if tonumber(zoneA) == tonumber(zoneB) then
                return nameA < nameB
            else
                return tonumber(zoneA) < tonumber(zoneB)
            end
        elseif tonumber(zoneA) then
            return true
        elseif tonumber(zoneB) then
            return false
        else
            if zoneA == zoneB then
                return nameA < nameB
            else
                return zoneA < zoneB
            end
        end
    end)

    return npcs
end

-- TAB Teleport
local TeleTab = Window:CreateTab("Teleport", nil) -- Title, Image
local TeleSection = TeleTab:CreateSection("Teleport")

-- Fungsi untuk mendapatkan daftar nama Zone beserta posisinya
local function getZoneListWithPositions()
    local zones = {}
    local zoneParent = workspace:WaitForChild("Zones")
    for _, zone in pairs(zoneParent:GetChildren()) do
        local position = zone.PrimaryPart and zone.PrimaryPart.Position or Vector3.new(0, 0, 0)
        table.insert(zones, {Name = zone.Name, Position = position})
    end

    -- Mengurutkan daftar zona
    table.sort(zones, function(a, b)
        if tonumber(a.Name) and tonumber(b.Name) then
            return tonumber(a.Name) < tonumber(b.Name)
        elseif tonumber(a.Name) then
            return true
        elseif tonumber(b.Name) then
            return false
        else
            return a.Name < b.Name
        end
    end)

    return zones
end

-- Inisialisasi dropdown dengan daftar Zone
local teleportZoneDropdown
local selectedZoneData = nil

local function createTeleportZoneDropdown()
    local zones = getZoneListWithPositions()
    local zoneNames = {}
    for _, zone in ipairs(zones) do
        table.insert(zoneNames, zone.Name)
    end

    teleportZoneDropdown = TeleTab:CreateDropdown({
        Name = "Zone To Teleport",
        SectionParent = TeleSection,
        Options = zoneNames,
        CurrentOption = "None",
        Callback = function(option)
            for _, zone in ipairs(zones) do
                if zone.Name == option then
                    selectedZoneData = zone
                    break
                end
            end
        end
    })
end

createTeleportZoneDropdown()

-- Fungsi untuk teleportasi karakter ke zona yang dipilih
local function teleportToSelectedZone()
    if selectedZoneData then
        local player = game.Players.LocalPlayer
        if player and player.Character and player.Character.PrimaryPart then
            player.Character:SetPrimaryPartCFrame(CFrame.new(selectedZoneData.Position))
        end
    end
end

-- Tombol untuk teleportasi
local teleportButton = TeleTab:CreateButton({
    Name = "Teleport",
    SectionParent = TeleSection,
    Callback = teleportToSelectedZone
})
